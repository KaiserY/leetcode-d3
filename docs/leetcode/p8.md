# 8. String to Integer (atoi)

## Example

```
Input: s = "  -42 word"
Output: -42
```

<P8 />

## Rust Solution

```rust
impl Solution {
    pub fn reverse(x: i32) -> i32 {
        let mut remainder = x;
        let mut output: i32 = 0;
        while remainder != 0 {
            if let Some(i) = output.checked_mul(10) {
                output = i;
                if let Some(i) = output.checked_add(remainder % 10) {
                    output = i;
                } else {
                    return 0;
                }
            } else {
                return 0;
            }
            remainder = remainder / 10;
        }

        return output;
    }
}
```
